[docs]def make_cpt(x):
    '''
    Convert numpy array x to conditional
    probability table
    '''
    res = x.copy()
    ranges = [range(s) for s in res.shape[:-1]]
    for s in product(*ranges):
        res[s] = softmax(res[s])

    return res
